    ```typescript
    import { computeCost, computeTranscriptionCost, type TextUsage } from "./pricing";

    type TrackCostArg =
      | number
      | undefined
      | { actualCost: number }
      | { usage: TextUsage }
      | { transcription: { provider: string; model: string; durationSeconds: number } };

    export async function trackCost(
      type: CostType,
      userId: string,
      arg?: TrackCostArg
    ): Promise<void> {
      // Resolve actual dollar cost
      let cost: number;
      if (arg === undefined) {
        cost = COST_ESTIMATES[type];
      } else if (typeof arg === "number") {
        cost = arg;
      } else if ("actualCost" in arg) {
        cost = arg.actualCost;
      } else if ("usage" in arg) {
        cost = computeCost(arg.usage);
      } else if ("transcription" in arg) {
        cost = computeTranscriptionCost(arg.transcription);
      } else {
        cost = COST_ESTIMATES[type];
      }

      // ... existing Redis increment logic using `cost` ...
    }
    ```
