  <files>app/lib/template/syntax-highlighter.tsx</files>
  <action>
    Extract the template syntax highlighting logic used in multiple components:

    ```typescript
    import { ReactNode } from 'react';

    /**
     * Highlights template syntax in a string.
     * Recognizes {{variable}}, [optional], and special keywords.
     *
     * @param text The template text to highlight
     * @returns Array of ReactNodes with highlighted spans
     */
    export function highlightTemplateSyntax(text: string): ReactNode[] {
      const parts: ReactNode[] = [];
      let lastIndex = 0;

      // Match template variables {{name}}, optionals [text], etc.
      const regex = /(\{\{[^}]+\}\})|(\[[^\]]+\])/g;
      let match;

      while ((match = regex.exec(text)) !== null) {
        // Add text before match
        if (match.index > lastIndex) {
          parts.push(text.slice(lastIndex, match.index));
        }

        // Add highlighted match
        if (match[1]) {
          // Variable: {{name}}
          parts.push(
            <span key={match.index} className="text-brand font-medium">
              {match[1]}
            </span>
          );
        } else if (match[2]) {
          // Optional: [text]
          parts.push(
            <span key={match.index} className="text-muted-foreground italic">
              {match[2]}
            </span>
          );
        }

        lastIndex = regex.lastIndex;
      }

      // Add remaining text
      if (lastIndex < text.length) {
        parts.push(text.slice(lastIndex));
      }

      return parts.length > 0 ? parts : [text];
    }

    /**
     * Component wrapper for syntax highlighting
     */
    export function TemplateSyntaxHighlight({ text }: { text: string }) {
      return <>{highlightTemplateSyntax(text)}</>;
    }
    ```

    Create a generic hook for dialog state management:

    ```typescript
    import { useState, useCallback } from 'react';

    /**
     * Generic dialog state management hook.
     * Handles open/close state and optional data for edit dialogs.
     *
     * @example
     * const { isOpen, data, open, close, openWith } = useDialog<User>();
     * // Open empty: open()
     * // Open with data: openWith(user)
     * // Close: close()
     */
    export function useDialog<T = void>() {
      const [isOpen, setIsOpen] = useState(false);
      const [data, setData] = useState<T | null>(null);

      const open = useCallback(() => {
        setIsOpen(true);
        setData(null);
      }, []);

      const openWith = useCallback((item: T) => {
        setData(item);
        setIsOpen(true);
      }, []);

      const close = useCallback(() => {
        setIsOpen(false);
        setData(null);
      }, []);

      const toggle = useCallback(() => {
        setIsOpen(prev => !prev);
      }, []);

      return {
        isOpen,
        data,
        open,
        openWith,
        close,
        toggle,
        setData,
        setIsOpen,
      };
    }

    // Type helper for edit dialogs
    export type DialogState<T> = ReturnType<typeof useDialog<T>>;
    ```

    This consolidates the common pattern of:
    - showDialog + setShowDialog
    - selectedItem + setSelectedItem
    - handleOpen + handleClose
