    /**
     * Detects imaging modality from transcribed text.
     * Searches for modality-specific keywords and patterns.
     */

    const MODALITY_PATTERNS: Record<string, RegExp[]> = {
      CT: [/\bct\b/i, /computed tomography/i, /cat scan/i],
      MRI: [/\bmri\b/i, /magnetic resonance/i, /\bmr\b/i],
      'X-Ray': [/\bx-?ray\b/i, /radiograph/i, /plain film/i],
      Ultrasound: [/ultrasound/i, /sonogram/i, /\bus\b/i, /doppler/i],
      PET: [/\bpet\b/i, /positron emission/i],
      Mammography: [/mammogra/i, /breast imaging/i],
      Fluoroscopy: [/fluoroscop/i, /\bgi\b.*study/i],
    };

    export function detectModality(text: string): string | null {
      const lowerText = text.toLowerCase();

      for (const [modality, patterns] of Object.entries(MODALITY_PATTERNS)) {
        if (patterns.some(pattern => pattern.test(lowerText))) {
          return modality;
        }
      }

      return null;
    }

    export function getAllModalities(): string[] {
      return Object.keys(MODALITY_PATTERNS);
    }
    ```

     * Detects body part from transcribed text.

    const BODY_PART_PATTERNS: Record<string, RegExp[]> = {
      Head: [/\bhead\b/i, /\bbrain\b/i, /\bcranial\b/i, /skull/i],
      Chest: [/\bchest\b/i, /thorax/i, /thoracic/i, /lung/i, /pulmonary/i],
      Abdomen: [/abdomen/i, /abdominal/i, /\bliver\b/i, /\bkidney/i, /spleen/i],
      Spine: [/spine/i, /spinal/i, /vertebra/i, /cervical/i, /lumbar/i, /thoracic spine/i],
      Extremity: [/extremit/i, /\barm\b/i, /\bleg\b/i, /\bhand\b/i, /\bfoot\b/i, /ankle/i, /wrist/i, /knee/i, /hip/i, /shoulder/i, /elbow/i],
      Pelvis: [/pelvis/i, /pelvic/i],
      Neck: [/\bneck\b/i, /cervical(?! spine)/i, /thyroid/i],
    };

    export function detectBodyPart(text: string): string | null {
      const lowerText = text.toLowerCase();

      for (const [bodyPart, patterns] of Object.entries(BODY_PART_PATTERNS)) {
        if (patterns.some(pattern => pattern.test(lowerText))) {
          return bodyPart;
        }
      }

      return null;
    }

    export function getAllBodyParts(): string[] {
      return Object.keys(BODY_PART_PATTERNS);
    }
    ```
